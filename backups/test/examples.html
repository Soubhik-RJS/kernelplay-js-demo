import { Game, Scene, Entity } from "kernelplay-js";
import { TransformComponent, CircleRenderComponent,
         Rigidbody2DComponent } from "kernelplay-js";

class BallScene extends Scene {
  init() {
    for (let i = 0; i < 18; i++) {
      const ball = new Entity(`Ball_${i}`);
      const r = Math.random() * 18 + 14;

      ball.addComponent("transform", new TransformComponent({
        position: {
          x: Math.random() * 760 + 20,
          y: Math.random() * 280 + 40,
        }
      }));

      ball.addComponent("renderer", new CircleRenderComponent({
        radius: r,
        color: `hsl(${Math.random()*360},70%,60%)`
      }));

      // Rigidbody with boundary bounce
      ball.addComponent("rigidbody2d", new Rigidbody2DComponent({
        velocity: {
          x: (Math.random() - 0.5) * 400,
          y: (Math.random() - 0.5) * 400,
        },
        useGravity: false,
        bounceWalls: true,  // built-in boundary bounce
      }));

      this.addEntity(ball);
    }
  }
}

class BouncingGame extends Game {
  init() {
    const scene = new BallScene("Bounce");
    this.sceneManager.addScene(scene);
    this.sceneManager.startScene("Bounce");
  }
}

new BouncingGame({
  width: 800, height: 360, fps: 60
}).start();
import { Game, Scene, Entity, ScriptComponent } from "kernelplay-js";
import { TransformComponent, BoxRenderComponent,
         Rigidbody2DComponent, ColliderComponent,
         Keyboard } from "kernelplay-js";

class PlayerController extends ScriptComponent {
  onStart() {
    this.rb = this.entity.getComponent("rigidbody2d");
    this.grounded = false;
    this.speed = 220;
  }

  update(dt) {
    // Horizontal
    if (Keyboard.isPressed("ArrowRight"))
      this.rb.velocity.x = this.speed;
    else if (Keyboard.isPressed("ArrowLeft"))
      this.rb.velocity.x = -this.speed;
    else
      this.rb.velocity.x *= 0.75;

    // Jump
    if (Keyboard.wasPressed("Space") && this.grounded) {
      this.rb.velocity.y = -420;
      this.grounded = false;
    }
  }

  onCollision(other) {
    if (other.tag === "platform") this.grounded = true;
  }
}

class PlatformerScene extends Scene {
  init() {
    // Ground
    this.createPlatform(0, 320, 800, 20, "ground");

    // Floating platforms
    [[100,240],[300,190],[480,150],[640,220]].forEach(([x, y]) => {
      this.createPlatform(x, y, 100, 14, "platform");
    });

    // Player
    const player = new Entity("Player");
    player.addComponent("transform", new TransformComponent({
      position: { x: 80, y: 100 }
    }));
    player.addComponent("renderer", new BoxRenderComponent("#6366f1", 28, 36));
    player.addComponent("rigidbody2d", new Rigidbody2DComponent({
      useGravity: true, mass: 1, drag: 0.02
    }));
    player.addComponent("collider", new ColliderComponent({ width:28, height:36 }));
    player.addComponent("script", new PlayerController());
    this.addEntity(player);
  }

  createPlatform(x, y, w, h, tag) {
    const p = new Entity(tag);
    p.tag = tag;
    p.addComponent("transform", new TransformComponent({ position: { x, y } }));
    p.addComponent("renderer", new BoxRenderComponent("#4f46e5", w, h));
    p.addComponent("collider", new ColliderComponent({ width: w, height: h }));
    this.addEntity(p);
  }
}
import { Game, Scene, Entity, ScriptComponent } from "kernelplay-js";
import { TransformComponent, Mouse } from "kernelplay-js";
import { ParticleEmitterComponent } from "kernelplay-js";

class ParticleScene extends Scene {
  init() {
    // Orbiting emitters
    const colors = ["#6366f1", "#ec4899", "#fbbf24"];
    colors.forEach((color, i) => {
      const emitter = new Entity(`Emitter_${i}`);
      emitter.addComponent("transform", new TransformComponent({
        position: { x: 400, y: 240 }
      }));
      emitter.addComponent("particles", new ParticleEmitterComponent({
        color,
        rate: 20,
        speed: { min: 40, max: 180 },
        lifetime: { min: 0.6, max: 1.4 },
        size: { min: 2, max: 7 },
        gravity: 120,
        spread: Math.PI * 2,
      }));
      emitter.addComponent("script", new class extends ScriptComponent {
        update(dt) {
          const t = this.entity.scene.time;
          const tf = this.entity.getComponent("transform");
          tf.position.x = 400 + Math.cos(t*.8 + i*2.1) * 240;
        }
      }());
      this.addEntity(emitter);
    });

    // Mouse-follow emitter
    const mouseEmitter = new Entity("MouseEmitter");
    mouseEmitter.addComponent("transform", new TransformComponent());
    mouseEmitter.addComponent("particles", new ParticleEmitterComponent({
      color: "white", rate: 30, speed: { min: 60, max: 200 },
      lifetime: { min: 0.4, max: 1 }, size: { min: 2, max: 5 },
    }));
    mouseEmitter.addComponent("script", new class extends ScriptComponent {
      update() {
        const tf = this.entity.getComponent("transform");
        tf.position.x = Mouse.x;
        tf.position.y = Mouse.y;
      }
    }());
    this.addEntity(mouseEmitter);
  }
}
import { Game, Scene, Entity, ScriptComponent } from "kernelplay-js";
import { TransformComponent, BoxRenderComponent,
         Rigidbody2DComponent, ColliderComponent } from "kernelplay-js";

// Zone component ‚Äî responds to trigger events
class ZoneEffect extends ScriptComponent {
  constructor(effect) { super(); this.effect = effect; }

  onTriggerEnter(other) {
    if (other.tag !== "player") return;
    console.log(`Entered: ${this.entity.name}`);

    const rb = other.getComponent("rigidbody2d");
    if (this.effect === "heal") {
      GameState.health = Math.min(100, GameState.health + 15);
    } else if (this.effect === "damage") {
      GameState.health = Math.max(0, GameState.health - 20);
    } else if (this.effect === "speed") {
      rb.velocity.x = 200;
      setTimeout(() => rb.velocity.x = 90, 1500);
    }
  }

  onTriggerExit(other) {
    console.log(`Left: ${this.entity.name}`);
  }
}

class TriggerScene extends Scene {
  init() {
    this.addZone(150, 100, 120, 100, "green", "heal", "Heal Zone");
    this.addZone(380, 130, 110, 90, "red", "damage", "Damage Zone");
    this.addZone(600, 100, 130, 110, "orange", "speed", "Speed Zone");

    const player = new Entity("Player");
    player.tag = "player";
    player.addComponent("transform", new TransformComponent({
      position: { x: 60, y: 160 }
    }));
    player.addComponent("renderer", new BoxRenderComponent("#6366f1", 28, 28));
    player.addComponent("rigidbody2d", new Rigidbody2DComponent({
      velocity: { x: 90, y: 0 }, useGravity: false
    }));
    player.addComponent("collider", new ColliderComponent({
      width: 28, height: 28
    }));
    this.addEntity(player);
  }

  addZone(x, y, w, h, color, effect, name) {
    const zone = new Entity(name);
    zone.addComponent("transform", new TransformComponent({ position:{x,y} }));
    zone.addComponent("renderer", new BoxRenderComponent(color, w, h));
    zone.addComponent("collider", new ColliderComponent({
      width: w, height: h, isTrigger: true  // ‚Üê trigger!
    }));
    zone.addComponent("script", new ZoneEffect(effect));
    this.addEntity(zone);
  }
}
import { Game, Scene, Entity, ScriptComponent } from "kernelplay-js";
import { TransformComponent, CircleRenderComponent, Mouse } from "kernelplay-js";

const G = 6000;

class GravityBody extends ScriptComponent {
  constructor(mass, fixed = false) {
    super();
    this.mass = mass;
    this.fixed = fixed;
    this.vx = 0; this.vy = 0;
  }

  update(dt) {
    if (this.fixed) return;
    const bodies = this.entity.scene.findEntitiesByTag("body");
    let ax = 0, ay = 0;

    for (const other of bodies) {
      if (other === this.entity) continue;
      const tf  = this.entity.getComponent("transform");
      const otf = other.getComponent("transform");
      const dx = otf.position.x - tf.position.x;
      const dy = otf.position.y - tf.position.y;
      const dist = Math.sqrt(dx*dx + dy*dy) + 0.1;
      const otherScript = other.getComponent("gravity");
      const f = G * otherScript.mass / (dist * dist);
      ax += (dx / dist) * f;
      ay += (dy / dist) * f;
    }

    const tf = this.entity.getComponent("transform");
    this.vx += ax * dt;
    this.vy += ay * dt;
    tf.position.x += this.vx * dt;
    tf.position.y += this.vy * dt;
  }
}

class GravityScene extends Scene {
  init() {
    // Sun
    this.spawnBody(400, 240, 22, 6000, true, "#fbbf24");

    // Click to add bodies
    this.on("click", (mx, my) => {
      const cx = 400, cy = 240;
      const dx = mx - cx, dy = my - cy;
      const d = Math.sqrt(dx*dx+dy*dy) || 1;
      const orbitSpeed = Math.sqrt(G * 6000 / d) * 0.85;
      this.spawnBody(mx, my, 6, 100, false, randomColor(), {
        vx: -(dy/d) * orbitSpeed,
        vy:  (dx/d) * orbitSpeed,
      });
    });
  }

  spawnBody(x, y, r, mass, fixed, color, vel = {vx:0,vy:0}) {
    const e = new Entity("body");
    e.tag = "body";
    e.addComponent("transform", new TransformComponent({ position:{x,y} }));
    e.addComponent("renderer", new CircleRenderComponent({ radius:r, color }));
    const script = new GravityBody(mass, fixed);
    script.vx = vel.vx || 0;
    script.vy = vel.vy || 0;
    e.addComponent("gravity", script);
    this.addEntity(e);
  }
}

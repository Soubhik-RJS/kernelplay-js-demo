player.js 
import { Entity, TransformComponent, BoxRenderComponent, 
         ScriptComponent, Keyboard } from "kernelplay-js";

class PlayerController extends ScriptComponent {
  update(dt) {
    const transform = this.entity.getComponent("transform");
    const speed = 200;
    
    if (Keyboard.isPressed("ArrowUp")) transform.position.y -= speed * dt;
    if (Keyboard.isPressed("ArrowDown")) transform.position.y += speed * dt;
    if (Keyboard.isPressed("ArrowLeft")) transform.position.x -= speed * dt;
    if (Keyboard.isPressed("ArrowRight")) transform.position.x += speed * dt;
  }
}

export class Player extends Entity {
  constructor(x, y) {
    super("Player");
    
    this.addComponent("transform", new TransformComponent({
      position: { x, y }
    }));
    this.addComponent("renderer", new BoxRenderComponent("blue"));
    this.addComponent("controller", new PlayerController());
  }
}
